hihocoder_1290_Demo_Day

题意:就是机器人能往右走就往右走，不能就往下走。往下走走不动了就再往右走。你能够把障碍物变成平地，也能把平地变成障碍物。问机器人从图的左上走到右下的话，最少需要修改图中的几个格子。
思路：DP problem。
dp[x][y][0]表示从从格子右边走的代价，dp[x][y][1]表示该格子往下边走的代价。
1、dp[1][1][0] = nums[1][1]==’.’?0:1; dp[1][1][1] = nums[1][1][0]+(nums[1][2]==’.’?1:0);(i==1,j==1)
2、dp[1][j][0] = dp[1][j-1][0]+(nums[1][j]==’.’?0:1);dp[1][j][1] = INT_MAX; (2<=j<=M)
3、dp[i][j][0] = dp[i][1][0]+(nums[i][1]==’.’?0:1);dp[i][1][0]=INT_MAX；(2<=i<=N)
output:dp[N][M]

code:

#include

#include

#include

#include

#include
using namespace std;

int N,M;
char a[100+5][100+5];
int dp[100+5][100+5][2];

/
dp[i][j][0]表示到达位置时，此时方向向右
dp[i][j][1]表示到达位置时，此时方向向下 /
int main()
{
while(scanf(“%d%d”,&N,&M)!=EOF)
{
for(int i=1;i<=N;++i)
scanf(“%s”,a[i]+1);
memset(dp,0,sizeof(dp));

dp[1][1][0] = a[1][1]=='.'?0:1;
dp[1][1][1] = dp[1][1][0] + ((2<=M)&&(a[1][2]=='.'?1:0));

for(int j=2;j<=M;++j)
{
    dp[1][j][0] = dp[1][j-1][0]+(a[1][j]=='.'?0:1);
    dp[1][j][1] = INT_MAX; 
}
for(int i=2;i<=N;++i)
{
    dp[i][1][0] = INT_MAX;
    dp[i][1][1] = dp[i-1][1][1]+(a[i][1]=='.'?0:1);
}

for(int i=2;i<=N;++i)for(int j=2;j<=M;++j)
{
    //左边过来的

    dp[i][j][0] = min(dp[i][j-1][0],dp[i][j-1][1]+((i+1<=N)&&(a[i+1][j-1]=='b'?0:1)))+(a[i][j]=='b'?1:0);
    //上边过来的
    dp[i][j][1] = min(dp[i-1][j][1],dp[i-1][j][0]+((j+1<=M)&&(a[i-1][j+1]=='b'?0:1)))+(a[i][j]=='b'?1:0);
}

int ans = dp[N][M][0]<dp[N][M][1]?dp[N][M][0]:dp[N][M][1];
cout<<ans<<endl;
}

return 0;

}

注意：
1、输入输出问题：对于char a[M][N],进行每行，从下标1开始输入scanf(“%s”,a[i]+1);或者每输入一行进行getchar();<剔除换行>
2、dp问题，注意每个格子的上一格，左一格都有两个方向需要分别保存。
2、不积跬步无以至千里，加油。
